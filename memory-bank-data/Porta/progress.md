# Возвращает: {"error": "Missing X-PORTA-TOKEN header"}

# С неверным токеном - доступ запрещен
curl -H "X-PORTA-TOKEN: wrong_token" -X GET "http://localhost:8111/meta" | jq
# Возвращает: {"error": "Invalid token"}
